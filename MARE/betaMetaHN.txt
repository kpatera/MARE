model
{
      for( i in 1 : Nstud ) {
      rA[i] ~ dbin(pA[i], nA[i])
      rB[i] ~ dbin(pB[i], nB[i])
      logit(pA[i]) <- mu[i]
      logit(pB[i]) <- mu[i] + delta[i]
      mu[i] ~ dt(0,pControl,Nstud) 
      delta[i] ~ dnorm(d, prec)
      }
      OR <- exp(d)
      d ~ dt(0,pControl,Nstud) 
      tau~dnorm(0,0.25)I(0,)
      tau.sq<-tau*tau
      prec<-1/(tau.sq)
      d.new ~dnorm(d,prec)
}
